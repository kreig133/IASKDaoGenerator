package ru.sbrf.iask.foundation.persistence.entity;

/**
 * Интерфейс, предоставляющий информацию о логине и идентификаторе (cuid) пользователя.
 *
 * @author Kuznetsov-VU
 * @author snetesanyi
 * @version 2.0
 */
public interface ICredentials {

    /**
     * Получает логин пользователя.
     *
     * @return Логин пользователя.
     */
    String getLogin();

    /**
     * Получает идентификатор пользователя (cuid).
     *
     * @return Идентификатор пользователя (cuid).
     */
    long getUserId();
}
