package ru.sbrf.iask.foundation.persistence.dao;

/**
 * Базовый класс для комманд БД.
 *
 * @param <R>
 *         тип возвращаемого результата команды.
 * @param <P>
 *         тип параметра запроса команды.
 * @author iskorodumov
 */
public abstract class AbstractDaoCommand<R, P> {
    /** объект запроса */
    private P request;

    /** Конструктор по умолчанию для команд без параметров запроса. */
    public AbstractDaoCommand() {
    }

    /**
     * Конструктор с параметрами.
     *
     * @param request
     *         объект запроса
     */
    public AbstractDaoCommand(P request) {
        this.request = request;
    }

    /**
     * Операция комманды.
     *
     * @return результат выполнения операции
     */
    public abstract R execute();

    /**
     * Get the command's request
     *
     * @return the command's request
     */
    public P getRequest() {
        return this.request;
    }
}
